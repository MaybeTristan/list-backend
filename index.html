    // ── Generate ───────────────────────────────────────────────────────────
    document.getElementById('generateBtn').addEventListener('click', async () => {
      const name     = document.getElementById('itemName').value.trim();
      const link     = document.getElementById('itemLink').value.trim();
      const img      = document.getElementById('itemImg').value.trim();
      const category = document.getElementById('itemCategory').value;
      const note     = document.getElementById('itemNote').value.trim();
      let   price    = document.getElementById('itemPrice').value.trim();
      const isNotAmazon = document.getElementById('notAmazonToggle').classList.contains('on');
      const asinRaw  = isNotAmazon ? '' : document.getElementById('asinInput').value.trim();
      const asin     = isNotAmazon ? null : (extractAsinFromText(asinRaw) || (/^[A-Z0-9]{10}$/i.test(asinRaw) ? asinRaw.toUpperCase() : null));

      if (!name || !link || !img) {
        await showModal({
          icon: '⚠️', title: 'Missing Required Fields',
          body: 'Please fill in the Item Name, Item Link, and Image URL before generating.',
          type: 'warning', buttons: [{ label: 'Got It', style: 'btn-primary' }]
        });
        return;
      }

      if (!asin && !price) {
        const res = await showModal({
          icon: '💰', title: 'No Price Set',
          body: 'This item has no ASIN (no live price) and no static fallback price. Would you like to set one now?',
          type: 'warning', input: true,
          inputPlaceholder: 'e.g. $14.99 — leave blank to skip',
          buttons: [{ label: 'Skip', style: 'btn-secondary' }, { label: 'Set Price', style: 'btn-primary' }]
        });
        if (res?.action === 'Set Price' && res.inputValue) {
          price = res.inputValue;
          document.getElementById('itemPrice').value = price;
        }
      }

      const finalPrice = price || '$0.00';
      const id         = generateId(name);
      const asinValue  = asin ? `"${asin}"` : 'null';
      const newLine    = `      {id:"${id}",name:"${name}",price:"${finalPrice}",asin:${asinValue},img:"${img}",link:"${link}",category:"${category}",prime:${isPrime},note:"${note}"},`;

      // Build full updated index.html by inserting new item line before the closing ];
      const updatedIndex = INDEX_TEMPLATE.replace(
        /(\s*\{id:"smartwool_medium_gray_socks"[^\n]+\},)/,
        '$1\n' + newLine
      );

      // Store for copy
      window._generatedCode = updatedIndex;
      window._newItemLine   = newLine;

      // Preview card
      document.getElementById('previewCard').innerHTML = `
        
        

          ${category}
          ${isPrime ? 'Prime ✓' : ''}
          ${note ? `${note}` : ''}
        

        

          
${name}

          
${asin ? '🟢 Live Price' : finalPrice}

          ${asin ? `
ASIN: ${asin}
` : '
No ASIN — Live Price Unavailable
'}
        
`;

      // Smooth reveal
      const out = document.getElementById('outputSection');
      out.classList.add('visible');
      setTimeout(() => out.scrollIntoView({ behavior: 'smooth', block: 'nearest' }), 100);
    });

    // ── Copy full file ─────────────────────────────────────────────────────
    document.getElementById('copyBtn').addEventListener('click', () => {
      if (!window._generatedCode) return;
      navigator.clipboard.writeText(window._generatedCode).then(() => {
        const btn = document.getElementById('copyBtn');
        btn.textContent = '✓ Copied!'; btn.classList.add('copied');
        setTimeout(() => { btn.textContent = 'Copy Full File'; btn.classList.remove('copied'); }, 2000);
      });
    });

    // ── Toggle full code view ──────────────────────────────────────────────
    document.getElementById('viewCodeBtn').addEventListener('click', () => {
      const codeEl  = document.getElementById('fullCodeWrap');
      const codeText = document.getElementById('fullCodeBlock');
      const btn     = document.getElementById('viewCodeBtn');
      const isOpen  = codeEl.classList.toggle('visible');
      btn.textContent = isOpen ? '▴ Hide Code' : '▾ View Full Code';
      if (isOpen && window._generatedCode) {
        codeText.textContent = window._generatedCode;
      }
    });

    // ── Not Amazon toggle ─────────────────────────────────────────────────
    let isNotAmazon = false;
    const notAmazonToggle = document.getElementById('notAmazonToggle');
    notAmazonToggle.addEventListener('click', () => {
      isNotAmazon = !isNotAmazon;
      notAmazonToggle.classList.toggle('on', isNotAmazon);
      const asinSection = document.getElementById('asinSection');
      asinSection.classList.toggle('hidden', isNotAmazon);
      // Update link label and price label
      const linkLabel = document.querySelector('label[for="itemLink"]') || document.querySelector('#itemLink').previousElementSibling;
      if (linkLabel) linkLabel.textContent = isNotAmazon ? 'Item Link *' : 'Item Link * — paste any link and we'll auto-detect the ASIN';
      document.getElementById('priceLabel').textContent = isNotAmazon
        ? 'Price *'
        : 'Fallback / Static Price (shown if no ASIN or price unavailable)';
      if (isNotAmazon) {
        clearAsinStatus();
        document.getElementById('asinInput').value = '';
        document.getElementById('itemLink').placeholder = 'https://www.rei.com/product/...';
      } else {
        document.getElementById('itemLink').placeholder = 'https://www.amazon.com/dp/...';
      }
    });

    // ── Reset ──────────────────────────────────────────────────────────────
    document.getElementById('resetBtn').addEventListener('click', async () => {
      const res = await showModal({
        icon: '↺', title: 'Reset Form', body: 'Clear all fields and start fresh?',
        buttons: [{ label: 'Cancel', style: 'btn-secondary' }, { label: 'Reset', style: 'btn-primary' }]
      });
      if (res?.action !== 'Reset') return;
      ['itemLink','itemName','itemNote','asinInput','itemPrice','itemImg'].forEach(id => document.getElementById(id).value = '');
      document.getElementById('itemCategory').value = 'Food';
      clearAsinStatus();
      document.getElementById('imgPreview').innerHTML = '
Enter an image URL above to preview
';
      document.getElementById('imgPreview').classList.remove('has-image');
      document.getElementById('outputSection').classList.remove('visible');
      isPrime = true; primeToggle.classList.add('on');
    });
